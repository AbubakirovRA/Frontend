Задание 3
Пользователь с клавиатуры вводит 3 числа, необходимо создать функцию, которая определяет максимальное значение среди этих чисел.

const getMax = (a, b, c) => {
  return (a > b && a > c) ? a : (b > a && b > c) ? b : c;
};

console.log(getMax(3, 9, 6));
